#version 460 core

layout(location=0) in vec2 position;
layout(location=1) in uint color_id;

const vec3 colors[3] = vec3[](
    vec3(1.0, 0.0, 0.0),
    vec3(0.0, 1.0, 0.0),
    vec3(0.0, 0.0, 1.0)
);

out vec3 fragmentColor;

void main()
{
    gl_Position = vec4(position, 0.0, 1.0);
    fragmentColor = colors[color_id];
}
