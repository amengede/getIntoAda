#version 460 core

const vec2 positions[3] = vec2[](
    vec2(-0.75, -0.75),
    vec2(0.75, -0.75),
    vec2(0.0, 0.75)
);

const vec3 colors[3] = vec3[](
    vec3(1.0, 0.0, 0.0),
    vec3(0.0, 1.0, 0.0),
    vec3(0.0, 0.0, 1.0)
);

out vec3 fragmentColor;

void main()
{
    gl_Position = vec4(positions[gl_VertexID], 0.0, 1.0);
    fragmentColor = colors[gl_VertexID];
}
